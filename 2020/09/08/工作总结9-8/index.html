    // let data = await this.db("jh_machine_product_type")

    // this.callback(data);

    let first_level,second_level;
    first_level=await this.db("jh_machine_product_type")
        .where({status: 1})
        .select("name as title", "id as value", "pid as group");
    second_level=await this.db("jh_machine_product_type")
        .where({status: 1,type : 2})
        .select("name as title", "id as value", "pid as group");
    let data={}
    if(first_level && second_level) {
        data.first_level = first_level;
        data.second_level = second_level;
        this.callback(data);
    } else {
        this.callback(false, {toast: '数据获取失败，请检查网络！'})
    }
}

    // console.log(edit)
    this.setState({edit,casual:edit.first_level_id}, () => {
        //弹窗展示

    })
}
