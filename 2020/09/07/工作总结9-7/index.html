    }

    //使用get获取classname相关信息
    @PropAction({auth: false})
    async get_class_name(get, {}) {
        let data = await this.db("jh_xiaobai_class")
            .select("classname as title", "cid as value")
        this.callback(data);
    }

    //添加用户
    @PropAction({})
    async do_add({}, {name,age,phone,gender,class_id}) {
        try {
            let model = await this.db('jh_xiaobai_test')
                .insert({
                    name,
                    age,
                    phone,
                    gender,
                    class_id,
                    time: time(),
                    // 每次需要修改都会添加修改时间
                    // u_time: strtotime(u_time)
                });
            this.record('添加成功!');
            let data = await this._listModel(model[0]);
            this.callback(data, {toast: "message.success.add"});
        } catch (ex) {
            console.log(ex)
            this.record('添加失败!', false, ex.message);
            this.callback(false, {toast: "message.error.add"});
        }
    }

                });

            let data = await this._listModel(id);

        } catch (ex) {
