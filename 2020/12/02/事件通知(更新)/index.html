    //核心

    tools, classnames,

    Table, TabPage, Lang, Dialog, Tree,
    Form, Page, Upload, Icon, Grid,
    BoxWidget, Input, Button, Prompt, Link, Loaders, Editor, Resource, Directive, Lock, Fix, Http, Excel, Image,
    $event
} from 'admin';

componentDidMount() {
	this.handlerGoodsBrandChange();
}

@$event('System.product_brand_change')
handlerGoodsBrandChange = () => {
	this.http
	.send(`${window.SERVER_URL}/${this.props.module}/goods/get_product_brand`)
	.success((data) => {
        this.setState({brand_list: data},()=>{
            this.refs.table && this.refs.table.refresh()
        })
    })
}
    
//XXXXX.js
/**
* 获取品牌信息
*/
@PropAction({auth: false})
async get_product_brand(get, {}) {
	let data = await this.db('jh_cx_product_brand')
		.where({status: 1})
		.select('name as title', 'id as value');
	this.callback(data);
}


//在品牌模块中找到添加成功/编辑成功后添加事件
