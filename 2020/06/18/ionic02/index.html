前端 框架

ionic

ionic 是 第三方的组件库, 主要用于开发手机端的UI.

可以结合vue, react,angular 三大框架进行使用

官网: https://ionicframework.com/

组件: https://ionicframework.com/docs/components

新建项目包, 到希望项目生成的目录下打开命令行

ionic start ionic-app blank

启动命令: 在项目目录下打开命令行

ionic s

弹出框组件

https://ionicframework.com/docs/api/alert

import { Component } from "@angular/core";

// 引入弹出框组件 AlertController
import { Platform, AlertController } from "@ionic/angular";
import { SplashScreen } from "@ionic-native/splash-screen/ngx";
import { StatusBar } from "@ionic-native/status-bar/ngx";

@Component({
  selector: "app-root",
  templateUrl: "app.component.html",
  styleUrls: ["app.component.scss"]
})
export class AppComponent {
  // 依赖注入: 在构造中声明需要什么类型的服务, 系统就会传入对应类型的服务
  constructor(
    private platform: Platform,
    private splashScreen: SplashScreen,
    private statusBar: StatusBar,
    private alertC: AlertController
  ) {
    this.initializeApp();
  }

  show() {
    // this.alertC.create({弹出框的配置}).then(弹出框=>{弹出框.弹出()})

    let options = {
      header: "东哥讲过Promiss吗?",
      message: "promiss是一个承诺, 承诺做完某件事之后, 继续做xxx, 小黄人例子",
      buttons: [
        {
          text: "讲过", //hanlder属性, 点击按钮后触发
          handler: () => {
            console.log("讲过");
          }
        },
        {
          text: "没讲过",
          handler: () => {
            console.log("没讲过");
          }
        }
      ]
    };

    // create: 创建
    // then: 创建之后, 执行xxx
    // create() 方法接收一个对象类型的配置项, 要配置弹出框中的所有内容
    // then(): 中需要手动让创建好的对话框进行显示
    // dialog: 就是创建出来的对话框, 这是人为定义的参数
    this.alertC.create(options).then(dialog => {
      // present(): 让弹出框出现
      dialog.present();
    });
  }

  initializeApp() {
    this.platform.ready().then(() => {
      this.statusBar.styleDefault();
      this.splashScreen.hide();
    });
  }
}

滑动组件

https://ionicframework.com/docs/api/slides

import { Component, ViewChild } from "@angular/core";

// 引入弹出框组件 AlertController
import { Platform, AlertController, IonSlides } from "@ionic/angular";
import { SplashScreen } from "@ionic-native/splash-screen/ngx";
import { StatusBar } from "@ionic-native/status-bar/ngx";

@Component({
  selector: "app-root",
  templateUrl: "app.component.html",
  styleUrls: ["app.component.scss"]
})
export class AppComponent {
  // <tag #s1></tag>
  // ionic默认脚手架中的 angular是8版本的, 目前最新的angular是9版本
  // 兼容问题: 9版本中: @ViewChild() 只要填写一个参数即可, 参数2有默认值
  // 8版本中, @ViewChild 必须要填写参数2
  @ViewChild("s1", { static: true }) slides: IonSlides;

  images = ["01.jpg", "02.jpg", "03.jpg", "04.jpg"];

  // 依赖注入: 在构造中声明需要什么类型的服务, 系统就会传入对应类型的服务
  constructor(
    private platform: Platform,
    private splashScreen: SplashScreen,
    private statusBar: StatusBar,
    private alertC: AlertController
  ) {
    this.initializeApp();
  }

  ngDoCheck(): void {
    // 默认的滚动操作会带有默认配置: 如果人为滚动了, 则自动滚动失效
    // this.slides.startAutoplay();

    //通过修改配置项, 让触碰失效 配置为假 即可解决
    this.slides.options = {
      autoplay: {
        loop: true, //自动滚动: true
        disableOnInteraction: false //触碰时自动滚动失效: false
      }
    };
  }

  show() {
    // this.alertC.create({弹出框的配置}).then(弹出框=>{弹出框.弹出()})

    let options = {
      header: "东哥讲过Promiss吗?",
      message: "promiss是一个承诺, 承诺做完某件事之后, 继续做xxx, 小黄人例子",
      buttons: [
        {
          text: "讲过", //hanlder属性, 点击按钮后触发
          handler: () => {
            console.log("讲过");
          }
        },
        {
          text: "没讲过",
          handler: () => {
            console.log("没讲过");
          }
        }
      ]
    };

    // create: 创建
    // then: 创建之后, 执行xxx
    // create() 方法接收一个对象类型的配置项, 要配置弹出框中的所有内容
    // then(): 中需要手动让创建好的对话框进行显示
    // dialog: 就是创建出来的对话框, 这是人为定义的参数
    this.alertC.create(options).then(dialog => {
      // present(): 让弹出框出现
      dialog.present();
    });
  }

  initializeApp() {
    this.platform.ready().then(() => {
      this.statusBar.styleDefault();
      this.splashScreen.hide();
    });
  }
}

import { Component, ViewChild } from "@angular/core";

// 引入弹出框组件 AlertController
import {
  Platform,
  AlertController,

@Component({
  selector: "app-root",
  templateUrl: "app.component.html",
  styleUrls: ["app.component.scss"]
})
export class AppComponent {
  names = [
    "宝马",
    "奥迪",
    "奔驰",
    "福特",
    "现代",
    "丰田",
    "本田",
    "吉利",
    "比亚迪",
    "特斯拉",
    "长城",
    "红旗",
    "DIOR",
    "Gucci",
    "CHANEL",
    "张嘿嘿",
    "嗯嗯嗯",
    "嘟嘟嘟",
    "路虎",
    "捷豹",
    "Object",
    "舒肤佳",
    "小黑瓶",
    "小棕瓶",
    "SKII"
  ];

  // 上拉触底时, 触发此方法
  loadData(obj) {
    console.log(obj);
    setTimeout(() => {
      this.names = this.names.concat(this.names);
      // 加载完毕后, 必须调用一个方法, 来告诉组件 可以监听下一次加载更多操作
      obj.target.complete();

      // 数据条数 > 100,  则组件无效
      if (this.names.length > 100) {
        obj.target.disabled = true;
      }
    }, 1000);
  }

  // 依赖注入: 在构造中声明需要什么类型的服务, 系统就会传入对应类型的服务
  constructor(
    private platform: Platform,
    private splashScreen: SplashScreen,
    private statusBar: StatusBar,
    private alertC: AlertController
  ) {
    this.initializeApp();
  }

  initializeApp() {
    this.platform.ready().then(() => {
      this.statusBar.styleDefault();
      this.splashScreen.hide();
    });
  }
}

生成带有Tabs栏的项目包

ionic start tabs-app tabs

新的命令行:

ionic中, 把组件通过功能的不同划分成了两个名字

  • component: 组件, 代表页面上的一个组成零件
  • page: 页, 代表整个页面
ng g c 组件名

ng g page 页面名

ng g p 管道
ng g pipe 管道

生成新的页面 tab4

ng g page tab4

路由制作src/app/tabs/tabs-routing.module.ts

import { NgModule } from "@angular/core";
import { RouterModule, Routes } from "@angular/router";
import { TabsPage } from "./tabs.page";

const routes: Routes = [
  {
    path: "tabs",
    component: TabsPage,
    children: [
      {
        path: "tab1",
        children: [
          {
            path: "",
            loadChildren: () =>
              import("../tab1/tab1.module").then(m => m.Tab1PageModule)
          }
        ]
      },
      {
        path: "tab2",
        children: [
          {
            path: "",
            loadChildren: () =>
              import("../tab2/tab2.module").then(m => m.Tab2PageModule)
          }
        ]
      },
      {
        path: "tab3",
        children: [
          {
            path: "",
            loadChildren: () =>
              import("../tab3/tab3.module").then(m => m.Tab3PageModule)
          }
        ]
      },
      {
        path: "tab4",
        children: [
          {
            path: "",
            loadChildren: () =>
              import("../tab4/tab4.module").then(m => m.Tab4PageModule)
          }
        ]
      },
      {
        path: "",
        redirectTo: "/tabs/tab1",
        pathMatch: "full"
      }
    ]
  },
  {
    path: "",
    redirectTo: "/tabs/tab1",
    pathMatch: "full"
  }
];

@NgModule({
  imports: [RouterModule.forChild(routes)],
  exports: [RouterModule]
})
export class TabsPageRoutingModule {}
