function find(arr) {
  for (let i = arr.length - 1; i >= 0; i--) {

题目2：在两个数组中，高效率找出相同的元素，放入新数组

// sort()
// 测试数组
var arr2 = [1, 22, 33, 55, 99, 8, 5, 6, 4]
var arr3 = [1, 2, 5, 9, 8, 44, 55, 6, 33]
// 分别遍历两个数组让其一一比较
function runk(x, y) {
  // 排序
  arr_x = x.sort((a, b) => a - b)
  arr_y = y.sort((a, b) => a - b)
  var res = [];
  for (let i = 0, j = 0; i < arr_x.length && j < arr_y.length;) {
    if (arr_x[i] > arr_y[j]) {
      j++;
    } else if (arr_x[i] < arr_y[j]) {
      i++;
    } else {
      res.push(arr_x[i]);
      i++, j++;
    }
  }
  console.log("两个数组中相同元素" + res)
}
runk(arr2, arr3)

题目3：英文首字母大写

// 测试字符串
var str1 = "HoW ArE YOu"

function lower(str) {
  // 全部转换成小写，以空格打散成数组
  arr = str.toLowerCase().split(" ");
  // console.log(str)
  for (let i = 0; i < arr.length; i++) {
    // 截取首字母，转大写拼接上单词后面部分
    arr[i] = arr[i].slice(0, 1).toUpperCase() + arr[i].slice(1)
    // 以空格拼接成字符串
    str = arr.join(" ")
  }
  console.log(str)
}
// 简洁版
function lower_less(str) {
  str = str.toLowerCase();
  // 匹配首字母
  str = str.replace(/\b[a-z]/ig, (keyword) => {
    return keyword.toUpperCase()
  })
  console.log(str)
}
lower(str1)
lower_less(str1)

题目4：查找字符出现的次数

//方法一 递归
function find_1(str) {
  // 定义变量
  var value = "";
  var index = 0;
  var obj = {};
  // 把字符以及出现的次数放入一个对象
  for (var i = 0; i < str.length; i++) {
    var key = str[i];
    if (!obj[key]) {
      obj[key] = 1
    } else {
      obj[key]++
    }
  }
  // 遍历数组取出数值最大的
  for (var key in obj) {
    if (index < obj[key]) {
      value = key;
      index = obj[key]
    }
  }
  console.log(value + "出现了" + index + "次")
}
//方法二 正则
function find_2(str) {
  // 定义变量
  var index = 0;
  var value = '';
  //字符串转数组
  var arr = str.split('');
  //数组排序，转成字符串
  str = arr.sort().join('');
  //正则"/1"代表后面内容是否相同
  var reg = /(\w)\1+/g;
  //判断
  str.replace(reg, function (val, item) {
    if (index < val.length) {
      index = val.length;
      value = item;
    }
  })
  console.log(value + "出现了" + index + "次")
}

题目6：递归–斐波拉契数列 第n项的和

// 递归
// 斐波拉契数列 第n项的和
function fib(n) {
  if (n == 1 || n == 2) {
    return 1
  }
  return fib(n - 1) + fib(n - 2);
}
console.log(fib(20))
//递归算法
function fib(n) {
  if (n < 3) {
    return 1;
  }
  //fib(7) = fib(6)+fib(5)
  //1 1 2 3 5 8 13
  return fib(n - 1) + fib(n - 2)
}
console.log(fib(40))
// for循环代替递归
function fib_f(n) {
  if (n < 3) return 1;
  else {
    var f1 = 1,
      f2 = 1,
      fn;
    for (var i = 3; i <= n; i++) {
      fn = f1 + f2;
      f1 = f2;
      f2 = fn;
    }
    return fn
  }
}
console.log(fib_f(50));
