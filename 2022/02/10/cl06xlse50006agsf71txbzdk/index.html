路由守卫

vue-router 提供的导航守卫主要用来通过跳转或取消的方式守卫导航。这里有很多方式植入路由导航中：全局的，单个路由独享的，或者组件级的。 

// route/index.js
// 全局
router.beforeEach((to,from,next)=>{
	// 判断路由是否需要守卫，通常根据路由中的meta来判断
	if(to.meta.auth){
		// 判断权限，比如登录权限
		if(isLogin){
			next()
		}else{
			// 跳转到登录页面，并且携带目标url地址
			next('/login?redirect='+to.fullPath)
		}
	}else{
		next()
	}
})
// 单个路由
const routes = [
  {
    path: '/',
    name: 'Home',
    component: Home,
    meta: {
      auth: true
    },
    beforeEnter(to, from, next) {
      console.table(to, from, next)
      // 判断是否已经登录
      if (window.isLogin) {
        next()
      } else {
        console.log("未登录",to.fullPath,)
        next('/login?redirect=' + to.fullPath)
      }
    },
  }
]

// 组件 about.vue
methods:{
  beforeRouteEnter(to, from) {
    // 在渲染该组件的对应路由被验证前调用
    // 不能获取组件实例 `this` ！
    // 因为当守卫执行时，组件实例还没被创建！
    next((vm) => {
      console.log(vm.data);
    });
  },
  beforeRouteUpdate(to, from) {
    // 在当前路由改变，但是该组件被复用时调用
    // 举例来说，对于一个带有动态参数的路径 `/users/:id`，在 `/users/1` 和 `/users/2` 之间跳转的时候，
    // 由于会渲染同样的 `UserDetails` 组件，因此组件实例会被复用。而这个钩子就会在这个情况下被调用。
    // 因为在这种情况发生的时候，组件已经挂载好了，导航守卫可以访问组件实例 `this`
    console.log(this.data)
  },
  beforeRouteLeave(to, from) {
    // 在导航离开渲染该组件的对应路由时调用
    // 与 `beforeRouteUpdate` 一样，它可以访问组件实例 `this`
  },
}

获取数据的时机

beforeRouteEnter 该组件被调用前，ajax请求数据等等

beforeRouteUpdate该组件被更新时对数据进行更新

导航完成以后created并且watch监听$route路由变化更新值

安装vuex

// store/index.js
import Vue from 'vue'
import Vuex from 'vuex'

Vue.use(Vuex)

export default new Vuex.Store({
  state: {
    isLogin:false
  },
  mutations: {
    login(state){
      console.log("调用")
      state.isLogin=true
    },
    logout(state){
      state.isLogin=false
    }
  },
  actions: {
    loginAsync({commit},username){
      return new Promise((resolve,reject)=>{
        setTimeout(()=>{
          if(username==="admin"){
            commit('login')
            resolve(true)
          }else{
            reject()
          }
        },1000)
      })
    }
  },
  // 严格模式，直接修改Vuex中的值会抛出异常
  strict: true
})

// 访问
// 值 this.$store.state
// 方法 mutations中的方法名 this.$store.commit('login')
// 异步方法 actions中的方法名 this.$store.dispatch('loginAsync',username)

插件

我们可以使用localStorage实现状态持久化，解决页面刷新状态丢失 我们使用Vuex插件，它接收store作为唯一参数，store.subscribe当store调用时就会调用

//初始化store中的值
export default (store)=>{
    // store 初始化，将储存在localStoreage中的状态还原
    if(localStorage){
        const user = JSON.parse(localStorage.getItem('user')).own
        if(user){
            store.commit('own/login',user.username)
        }
    }
    //监听变化,储存在localStoreage中
    store.subscribe((mutation,state)=>{
        console.log(mutation,state)
        if(mutation.type==='own/login'){
            const user=JSON.stringify(state)
            console.log(user)
            localStorage.setItem('user',user)
        }else{
            localStorage.removeItem('user')
        }
    })
}
