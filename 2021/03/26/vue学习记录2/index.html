Vue

一、Axios

- 安装

npm install axios --save

- 在需要**.vue中引入,并使用

import axios  from 'axios '

- 请求方式

1. post请求
axios.post('/user', {
	firstName: 'One',
	lastName: 'Second'
})
.then(function (response) {
	console.log(response);
})
.catch(function (error) {
	console.log(error);
});
2.get请求
function getUserID() {
	return axios.get('/user/123');
}
 
function getUserCompelete() {
	return axios.get('/user/123/compelete');
}
 
axios.all([getUserID(), getUserCompelete()])
	.then(axios.spread(function (user_id, uesr_com) {
//两个请求现已完成
}));

二、vue-resource

- 安装

npm install vue-resource --save

- 在main.js中引入,并使用

import VueResource from 'vue-resource'
Vue.use(VueResource)

- 请求方式

1. post请求
getInfo(){
    this.$http.get('http:***********').then(result=>{
        console.log(result.body)
    })
 }
2.get请求
postInfo(){
//手动发起的post请求，默认没有表单格式，所以，有的服务器处理不了，通过post方法的第三个参数{ emulateJSON: true }设置提交的内容类型为普通表单数据格式。
     this.$http.post('http:*******',{},{ emulateJSON: true })
     .then(result=>{console.log(result.body)})
}
3.json请求
getInfo(){// JSONP形式从服务器获取数据
    this.$http.json('http:*********').then(result=>{
        console.log(result.body)
    })
 }
// 请求方式总计
get(url, [options])
head(url, [options])
elete(url, [options])
jsonp(url, [options])
post(url, [body], [options])
put(url, [body], [options])
patch(url, [body], [options])
// 注意：
// 1.url，请求地址。可被options对象中url属性覆盖。
// 2.body（可选，字符串或对象），要发送的数据，可被options对象中的body属性覆盖。
// 3.options 请求选项对象
