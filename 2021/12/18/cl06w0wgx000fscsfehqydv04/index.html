// 找到所有img
let images = document.querySelectorAll('img')
// 是否进入到可视区域
const callback = entries => {
    console.log(entries, "触发")
    entries.forEach(option => {
        if (option.isIntersecting) {
            const image = option.target;
            const data_src = image.getAttribute('data-src');
            image.setAttribute('src', data_src)
            observer.unobserve(image);
        }
    })
}
const observer = new IntersectionObserver(callback)
// 考虑兼容性
if (observer) {
    images.forEach(v => {
        observer.observe(v)
    })
} else {
    // 事件监听
    window.addEventListener('scroll', (e) => {
        images.forEach(image => {
            const imageTop = image.getBoundingClientRect().top;
            if (imageTop < window.innerHeight) {
                const data_src = image.getAttribute('data-src');
                image.setAttribute('src', data_src)
            }
        })
    })
}

js执行：微任务+宏任务

javascript是单线程处理函数，一个任务就是指计划由标准机制来执行的任何 JavaScript，如程序的初始化、事件触发的回调等。 除了使用事件，你还可以使用 setTimeout() 或者 setInterval() 来添加任务。

任务队列和微任务队列的区别很简单，但却很重要：

  • 当执行来自任务队列中的任务时，在每一次新的事件循环开始迭代的时候运行时都会执行队列中的每个任务。在每次迭代开始之后加入到队列中的任务需要在下一次迭代开始之后才会被执行.
  • 每次当一个任务退出且执行上下文为空的时候，微任务队列中的每一个微任务会依次被执行。不同的是它会等到微任务队列为空才会停止执行——即使中途有微任务加入。换句话说，微任务可以添加新的微任务到队列中，并在下一个任务开始执行之前且当前事件循环结束之前执行完所有的微任务。

function debounce(fun,delay){
    let timer;
    return function () {
        clearTimeout(timer)
        timer = setTimeout(fun, delay)
    }
}

class

class Father {
    constructor(name,age){
        this.name=name,
        this.age=age
    }
    //定义静态方法
    static welcome(){
        document.write('</br> 自我介绍')
    }
    show(){
        document.write("姓名:",this.name)
    }
    // 属性 访问时不要加()
    set other(value){
        this.value=value
    }
    get other(){
        return this.name+=this.value
    }
}

const father=new Father('xm','24');

class Son extends Father {
    // 子类必须要写super
    constructor(name,age,father){
        super(name,age);
        this.father=father
    };
    show(){
        document.write("姓名:",this.name,"爹:",this.father)
    }
}
Father.welcome()
father.show()
Son.welcome()
const son =new Son('xxm','1','xm')
son.show()
father.other="我是爹"
son.other="我是儿子"
console.log(son.other)
console.log(father.other)

            arr1.push(...child_arr)
        } else {
            arr1.push(data[i].name)

}
