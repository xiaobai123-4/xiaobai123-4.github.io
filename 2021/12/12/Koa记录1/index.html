    const Koa = require('koa'),
      //路由
      router = require('koa-router')(),
      //路径模块
      path = require('path')
    //应用模板中间件
    views = require('koa-views'),
      //静态资源中间件
      static = require('koa-static'),
      common = require('./module/common.js'),
      // 配置中间件bodyParser
      bodyParser = require('koa-bodyparser'),
      //模板引擎art-template
      render = require('koa-art-template');

    const app = new Koa()
    //第三方中间件 托管静态资源
    app.use(static('./static'
      // path.join(__dirname,'public')
    ))
    // 配置中间件bodyParser
    app.use(bodyParser())

    //第三方中间件 配置模板引擎中间件
    // 第一种  html
    // app.use(views('views',{map:{html:'ejs'}}))
    // 第二种  ejs
    app.use(views('views', {
      // 应用模板引擎
      extension: 'ejs'
    }))

    //art-template模板引擎
    render(app, {
      root: path.join(__dirname, 'views'),//视图位置
      extname: '.ejs',//后缀名
      debug: true//是否开启调试
    });

    app.use(async (ctx) => {
      let list = {
        title: '标题',
        info: [
          { name: "Tony", age: 15 },
          { name: "Mack", age: 19 },
          { name: "Tom", age: 17 },]
      }
      await ctx.render('art-template', { list });
    });


    router.get('/', async (ctx) => {
      let title = "你好"
      await ctx.render('index', { title })
    })

    router.get('/test1', async (ctx) => {
      await ctx.render('text-from')
    })

    router.post('/doAdd', async (ctx) => {
      //koa中原生请求①
      //var data=await common.getPostData(ctx);
      //②使用配置中间件
      var data = ctx.request.body
      //返回页面
      ctx.body = data
    })


    //应用级中间件 在访问所有路由之前都会调用
    // app.use(async (ctx,next)=>{
    //   ctx.body="应用级中间件";
    //   let time=new Date()
    //   console.log(time)
    //   await next()
    // })

    //错误处理中间件
    app.use(async (ctx, next) => {
      await next();
      if (ctx.status == 404) {
        ctx.body = "这是404页面"
      } else {
        console.log(ctx.url)
      }
    })

    //路由级中间件 匹配路由时继续向下匹配
    router.get('/news', async (ctx, next) => {
      // ctx.body="新闻get1"
      console.log("新闻get1")
      await next()
    });

    router.get('/', (ctx, next) => {
      ctx.body = "Hello Koa"
    });

    router.post('/news', (ctx, next) => {
      ctx.body = "新闻post"
    });

    router.get('/user/:id', (ctx, next) => {
      ctx.body = "user"
      console.log(ctx.params)
    })

    router.get('/news', (ctx, next) => {
      ctx.body = "新闻get"
      console.log(ctx.query)
    })

    app.use(router.routes())//启动路由
    app.use(router.allowedMethods())

    app.listen(3000, () => {
      console.log("post:3000 starting")
    })
